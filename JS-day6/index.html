1. Introduction to Objects in JavaScript ES5

Objects are one of the fundamental data types in JavaScript. They are collections of key-value pairs where keys are strings (properties) and values can be any data type including functions (methods).

What is an Object?

An object is a container that stores related data and functionality together. Think of it as a real-world entity with characteristics (properties) and behaviors (methods).

// Real-world analogy: A car object
var car = {
    brand: "Toyota",        // Property
    model: "Camry",         // Property
    year: 2020,             // Property
    start: function() {     // Method
        console.log("Car is starting...");
    }
};

console.log("Car details:", car);

2. Object Creation Methods

Method 1: Object Literal (Most Common)

The simplest and most common way to create objects in JavaScript.

var person = {
    name: "John Doe",
    age: 30,
    city: "New York",
    greet: function() {
        return "Hello, I'm " + this.name;
    }
};

console.log(person.name);     // "John Doe"
console.log(person.greet());  // "Hello, I'm John Doe"

Method 2: Object() Constructor

Creates an empty object and then adds properties dynamically.

var student = new Object();
student.name = "Alice";
student.grade = "A";
student.subject = "Math";

// Alternative syntax
var student2 = Object();  // Same as new Object()
student2.name = "Bob";

Method 3: Object.create()

Creates a new object with the specified prototype object and properties.

// Create object with null prototype (no inherited properties)
var teacher = Object.create(null);
teacher.name = "Mr. Smith";

// Create object with Object.prototype
var employee = Object.create(Object.prototype);
employee.id = 123;

3. Properties and Methods

Understanding Properties

Properties are key-value pairs that store data in objects. They can contain any JavaScript data type.

var laptop = {
    // String property
    brand: "Dell",
    
    // Number property
    price: 899.99,
    
    // Boolean property
    inStock: true,
    
    // Array property
    features: ["SSD", "8GB RAM", "Intel i5"],
    
    // Object property
    specs: {
        processor: "Intel i5",
        memory: "8GB",
        storage: "256GB SSD"
    },
    
    // Function property (method)
    getDescription: function() {
        return this.brand + " laptop - $" + this.price;
    }
};

Methods vs Functions

Methods are functions that belong to objects and can access object properties using 'this'.

var calculator = {
    result: 0,
    
    add: function(number) {
        this.result += number;
        return this;  // Method chaining
    },
    
    multiply: function(number) {
        this.result *= number;
        return this;
    },
    
    getValue: function() {
        return this.result;
    },
    
    reset: function() {
        this.result = 0;
        return this;
    }
};

// Method chaining example
var result = calculator.add(5).multiply(3).getValue(); // 15

4. Accessing Object Members

Dot Notation

The most common way to access object properties when you know the property name at compile time.

var book = {
    title: "1984",
    author: "George Orwell",
    pages: 328
};

console.log(book.title);   // "1984"
console.log(book.author);  // "George Orwell"

Bracket Notation

Used when property names contain special characters, spaces, or are stored in variables.

var product = {
    "product-name": "Smartphone",
    "unit price": 599.99,
    "in-stock": true
};

// Accessing properties with special characters
console.log(product["product-name"]);  // "Smartphone"
console.log(product["unit price"]);    // 599.99

// Dynamic property access
var propertyName = "in-stock";
console.log(product[propertyName]);    // true

Property Existence Checking

var user = {
    username: "john_doe",
    email: "john@example.com"
};

// Using 'in' operator
console.log("username" in user);     // true
console.log("password" in user);     // false

// Using hasOwnProperty (checks own properties, not inherited)
console.log(user.hasOwnProperty("email"));     // true
console.log(user.hasOwnProperty("toString"));  // false

// Using undefined check
console.log(user.username !== undefined);  // true
console.log(user.password !== undefined);  // false

5. The 'this' Keyword

Understanding 'this'

The 'this' keyword refers to the object that is currently executing the function. In object methods, 'this' refers to the object that owns the method.

var bankAccount = {
    accountNumber: "123456789",
    balance: 1000,
    
    deposit: function(amount) {
        if (amount > 0) {
            this.balance += amount;  // 'this' refers to bankAccount
            return "Deposited $" + amount + ". New balance: $" + this.balance;
        }
        return "Invalid amount";
    },
    
    getAccountInfo: function() {
        return "Account " + this.accountNumber + " has $" + this.balance;
    }
};

console.log(bankAccount.deposit(500));  // Updates bankAccount.balance
console.log(bankAccount.getAccountInfo());

Common 'this' Pitfalls

var counter = {
    count: 0,
    increment: function() {
        this.count++;
        console.log("Count:", this.count);
    }
};

// Direct call - 'this' refers to counter
counter.increment();  // Count: 1

// Assigning method to variable - 'this' context is lost
var incrementFunc = counter.increment;
incrementFunc();      // 'this' might refer to window/global object

// Solution: Using call() or apply()
incrementFunc.call(counter);  // 'this' explicitly set to counter

6. Constructor Functions

Creating Constructor Functions

Constructor functions are regular functions used with the 'new' keyword to create multiple objects with the same structure.

// Constructor function (note the capital letter convention)
function Car(brand, model, year) {
    this.brand = brand;
    this.model = model;
    this.year = year;
    this.mileage = 0;
    
    this.drive = function(miles) {
        this.mileage += miles;
        return this.brand + " driven " + miles + " miles. Total: " + this.mileage;
    };
    
    this.getAge = function() {
        var currentYear = new Date().getFullYear();
        return currentYear - this.year;
    };
}

// Creating instances
var car1 = new Car("Toyota", "Camry", 2019);
var car2 = new Car("Honda", "Civic", 2021);

console.log(car1.drive(100));  // Toyota driven 100 miles. Total: 100
console.log(car2.getAge());    // Age based on current year

What Happens with 'new'

When you use 'new' with a constructor function:

  1. A new empty object is created
  2. The new object's prototype is set
  3. 'this' is bound to the new object
  4. The constructor function is executed
  5. The new object is returned (unless constructor returns an object)
// Without 'new' - problematic
function Person(name) {
    this.name = name;  // 'this' refers to global object
}

var person1 = Person("John");  // Don't do this!
console.log(person1);          // undefined
console.log(window.name);      // "John" (in browser)

// With 'new' - correct
var person2 = new Person("Jane");
console.log(person2.name);     // "Jane"

7. Prototype and Inheritance

Understanding Prototype

Every function in JavaScript has a prototype property. When creating objects with constructor functions, the prototype is used to share methods among all instances.

function Animal(name, species) {
    this.name = name;
    this.species = species;
}

// Adding methods to prototype (shared by all instances)
Animal.prototype.speak = function() {
    return this.name + " makes a sound";
};

Animal.prototype.getInfo = function() {
    return this.name + " is a " + this.species;
};

Animal.prototype.isAlive = true;  // Shared property

var dog = new Animal("Buddy", "Dog");
var cat = new Animal("Whiskers", "Cat");

console.log(dog.speak());     // "Buddy makes a sound"
console.log(cat.getInfo());   // "Whiskers is a Cat"
console.log(dog.isAlive);     // true

Prototype vs Instance Properties

function Student(name) {
    this.name = name;           // Instance property
    this.grades = [];           // Instance property
}

Student.prototype.school = "Default School";  // Prototype property

Student.prototype.addGrade = function(grade) {
    this.grades.push(grade);
};

var student1 = new Student("Alice");
var student2 = new Student("Bob");

// Instance properties are unique to each object
student1.addGrade(95);
student2.addGrade(87);

console.log(student1.grades);  // [95]
console.log(student2.grades);  // [87]

// Prototype properties are shared
console.log(student1.school);  // "Default School"
console.log(student2.school);  // "Default School"

// Changing prototype affects all instances
Student.prototype.school = "New School";
console.log(student1.school);  // "New School"
console.log(student2.school);  // "New School"

Inheritance in ES5

// Parent constructor
function Vehicle(make, model) {
    this.make = make;
    this.model = model;
    this.speed = 0;
}

Vehicle.prototype.accelerate = function(increment) {
    this.speed += increment;
    return this.make + " " + this.model + " speed: " + this.speed + " mph";
};

Vehicle.prototype.brake = function() {
    this.speed = 0;
    return this.make + " " + this.model + " has stopped";
};

// Child constructor
function Motorcycle(make, model, engineSize) {
    Vehicle.call(this, make, model);  // Call parent constructor
    this.engineSize = engineSize;
    this.type = "Motorcycle";
}

// Set up inheritance
Motorcycle.prototype = Object.create(Vehicle.prototype);
Motorcycle.prototype.constructor = Motorcycle;

// Override parent method
Motorcycle.prototype.accelerate = function(increment) {
    this.speed += increment * 1.5;  // Motorcycles accelerate faster
    return this.make + " " + this.model + " (motorcycle) speed: " + this.speed + " mph";
};

// Add new method
Motorcycle.prototype.wheelie = function() {
    return this.make + " " + this.model + " is doing a wheelie!";
};

var bike = new Motorcycle("Harley", "Davidson", "1200cc");
console.log(bike.accelerate(30));  // Uses overridden method
console.log(bike.wheelie());       // Uses new method
console.log(bike.brake());         // Uses inherited method

8. Built-in Object Methods

Object.keys()

Returns an array of an object's own enumerable property names.

var person = {
    firstName: "John",
    lastName: "Doe",
    age: 30
};

var keys = Object.keys(person);
console.log(keys);  // ["firstName", "lastName", "age"]

Object.getOwnPropertyNames()

Returns an array of all own properties (enumerable and non-enumerable).

var obj = {};
Object.defineProperty(obj, 'hidden', {
    value: 'secret',
    enumerable: false
});
obj.visible = 'public';

console.log(Object.keys(obj));              // ["visible"]
console.log(Object.getOwnPropertyNames(obj)); // ["hidden", "visible"]

Object.defineProperty()

Defines a new property or modifies an existing property with specific attributes.

var product = {};

Object.defineProperty(product, 'price', {
    value: 99.99,
    writable: false,     // Cannot be changed
    enumerable: true,    // Shows up in for...in
    configurable: false  // Cannot be deleted or reconfigured
});

console.log(product.price);  // 99.99
product.price = 149.99;      // Silently fails (strict mode would throw error)
console.log(product.price);  // Still 99.99

9. Advanced Object Concepts

Object Property Descriptors

Every property has a descriptor that defines its behavior.

var obj = { name: "Test" };

// Get property descriptor
var descriptor = Object.getOwnPropertyDescriptor(obj, 'name');
console.log(descriptor);
// { value: "Test", writable: true, enumerable: true, configurable: true }

// Define property with custom descriptor
Object.defineProperty(obj, 'id', {
    value: 123,
    writable: false,
    enumerable: false,
    configurable: false
});

// Try to modify
obj.id = 456;           // Fails silently
console.log(obj.id);    // Still 123

// Check if property is enumerable
for (var prop in obj) {
    console.log(prop);  // Only "name" appears, not "id"
}

Getters and Setters

var temperature = {
    celsius: 0,
    
    get fahrenheit() {
        return (this.celsius * 9/5) + 32;
    },
    
    set fahrenheit(value) {
        this.celsius = (value - 32) * 5/9;
    }
};

temperature.celsius = 25;
console.log(temperature.fahrenheit);  // 77

temperature.fahrenheit = 86;
console.log(temperature.celsius);     // 30

Object Sealing and Freezing

var obj = { name: "Test", value: 42 };

// Object.seal() - prevents adding/removing properties, allows modification
Object.seal(obj);
obj.name = "Modified";    // Works
obj.newProp = "New";      // Fails silently
console.log(obj.name);    // "Modified"

var obj2 = { name: "Test", value: 42 };

// Object.freeze() - prevents all changes
Object.freeze(obj2);
obj2.name = "Modified";   // Fails silently
obj2.newProp = "New";     // Fails silently
console.log(obj2.name);   // Still "Test"

// Check status
console.log(Object.isSealed(obj));   // true
console.log(Object.isFrozen(obj2));  // true
